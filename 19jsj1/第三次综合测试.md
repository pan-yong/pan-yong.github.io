```c
#include<stdio.h>
void fun(char s[])
{
    int i = 0, j = 0;
    for (; s[i]!= '\0'; i++)
        if (s[i] >= '0' & s[i] <= '9'){
            s[j] = s[i];
            j--;
        }
    s[j] = '\0';
}
int main()
{
    char s[20] = "2022GoKao888";
    fun(s);
    printf("%s", s);
}
```
